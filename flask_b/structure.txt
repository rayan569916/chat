│
│   ├── services/         # Business logic (separate from routes)
│   │   └── user_service.py
│   │   └── chat_service.py
│
│   ├── schemas/          # Marshmallow schemas for validation/serialization
│   │   └── user_schema.py
│
│   ├── utils/            # Helpers (JWT, email, etc.)
